CREATE DATABASE IF NOT EXISTS agendamento;
USE agendamento;


CREATE TABLE Cliente (
    IdCliente BIGINT PRIMARY KEY,
    NomeCliente VARCHAR(255),
    TelefoneCliente VARCHAR(255)
);


CREATE TABLE Funcionario (
    IdFuncionario BIGINT PRIMARY KEY,
    NomeFuncionario VARCHAR(255),
    especialidade VARCHAR(255),
    TelefoneFuncionario VARCHAR(255),
    email VARCHAR(255),
    HorarioTrabalhoInicio TIME,
    HorarioTrabalhoFim TIME,
    DiasDeTrabalho VARCHAR(255)
);


CREATE TABLE Servico (
    IdServico BIGINT PRIMARY KEY,
    NomeServico VARCHAR(255),
    duracao INT,
    ValorServico DOUBLE
);


CREATE TABLE Agendamento (
    IdAgendamento BIGINT PRIMARY KEY,
    IdCliente BIGINT,
    TelefoneCliente VARCHAR(255),
    IdFuncionario BIGINT,
    DataServico VARCHAR(255),
    DataAgendamento DATE,
    HorarioReservado VARCHAR(255),
    IdServicoAgendado BIGINT,
    ValorServico DOUBLE,
    FOREIGN KEY (IdCliente) REFERENCES Cliente(IdCliente) ON DELETE CASCADE,
    FOREIGN KEY (IdFuncionario) REFERENCES Funcionario(IdFuncionario) ON DELETE CASCADE,
    FOREIGN KEY (IdServicoAgendado) REFERENCES Servico(IdServico) ON DELETE CASCADE
);
